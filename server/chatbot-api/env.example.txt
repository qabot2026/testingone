# =============================================================================
# Resend (HTTPS email API) — recommended on Railway Free / Trial / Hobby plans
# =============================================================================
# Railway blocks outbound SMTP on lower tiers, so Gmail / Office SMTP just
# hangs. Resend uses plain HTTPS, so it works on every tier.
#
# When RESEND_API_KEY is set, every transactional email in this codebase
# (contact-form notify, visitor ack, appointment ack, dashboard sign-in)
# is routed through Resend automatically. SMTP_* is ignored.
# Unset RESEND_API_KEY to fall back to SMTP.
#
# 1) Sign up at https://resend.com (free tier: 3,000/month, 100/day)
# 2) Create an API key:  https://resend.com/api-keys     → starts with "re_"
# 3) Paste below, redeploy. (Optional) verify a domain at
#    https://resend.com/domains so you can use a custom From and send to any
#    recipient. Without a verified domain, Resend only delivers to YOUR
#    signed-up email — set RESEND_FROM=onboarding@resend.dev in that case.
# If you still have MAIL_FROM=a@gmail.com for old SMTP, keep RESEND_FROM set:
# Resend sends prefer RESEND_FROM over MAIL_FROM (Gmail is rejected as From until you verify a domain).
# RESEND_API_KEY=re_xxxxxxxxxxxxxxxxxxxxxxxxxxxx
# RESEND_FROM=onboarding@resend.dev          # only needed until you verify a domain
# RESEND_BASE_URL=https://api.resend.com     # rarely changed
# RESEND_SEND_TIMEOUT_MS=25000               # 5000..120000
# RESEND_DEBUG=                              # 1 to log each send id

# SMS OTP — server/chatbot-api/lib/sms-otp.mjs (provider-agnostic core)
# =============================================================================
# Endpoints added by mountSmsOtpRoutes(app) in index.mjs:
#   POST /api/sms-otp/send    { "mobile": "9999999999" }
#   POST /api/sms-otp/verify  { "mobile": "9999999999", "code": "123456" }
#   GET  /api/sms-otp/health
#
# Active provider (default: msg91). Add more providers by dropping a file in
# server/chatbot-api/lib/sms-providers/ and registering it in
# the PROVIDERS map at the top of lib/sms-otp.mjs.
# SMS_OTP_PROVIDER=msg91
#
# ---- Provider: MSG91 (lib/sms-providers/msg91.mjs) --------------------------
# Required:
#   1) MSG91_AUTHKEY — Auth Key from https://control.msg91.com/app/#/api
#   2) MSG91_TEMPLATE_ID — DLT-approved template id. Your template MUST contain
#      ##OTP## where the code should appear, e.g.
#      "Your verification code is ##OTP##. Valid for 5 minutes. - XYZCLN"
# MSG91_AUTHKEY=your_msg91_authkey_here
# MSG91_TEMPLATE_ID=your_dlt_approved_template_id_here
#
# Optional:
# MSG91_SENDER_ID=          # usually already baked into the DLT template
# MSG91_COUNTRY_CODE=91     # default "91" (India). MSG91 expects <cc><mobile>, no '+'
# MSG91_BASE_URL=https://control.msg91.com   # override only for staging
#
# ---- Core OTP behaviour (applies to every provider) ------------------------
# SMS_OTP_LENGTH=6                # 4..8, default 6
# SMS_OTP_TTL_SECONDS=300         # 30..900, default 300 (5 minutes)
# SMS_OTP_MAX_ATTEMPTS=5          # 1..20, default 5 verify failures before lockout
# SMS_OTP_RESEND_COOLDOWN_S=30    # 0..600, default 30 seconds between sends per mobile
# SMS_OTP_IP_LIMIT=10             # 1..200, default 10 sends per IP per window
# SMS_OTP_IP_WINDOW_S=600         # 60..3600, default 600 (10 min)
#
# ---- Storage (Firestore primary, in-memory fallback) -----------------------
# Firestore is used automatically when FIREBASE_SERVICE_ACCOUNT_JSON is set
# (so codes survive Railway restarts + multi-instance). To force in-memory only:
# SMS_OTP_DISABLE_FIRESTORE=1
# SMS_OTP_FIRESTORE_COLLECTION=sms_otp_codes
#
# Optional audit trail (one Firestore doc per send/verify outcome):
# SMS_OTP_AUDIT_TO_FIRESTORE=1
# SMS_OTP_AUDIT_COLLECTION=sms_otp_audit
#
# After deploy, verify configuration in the browser:
#   GET https://YOUR-API.railway.app/api/sms-otp/health
# `provider_ready: false` means a required env var is missing — the
# `provider_missing_env` field lists exactly which one(s).

# Customization dashboard — server/chatbot-api/lib/dashboard/index.mjs
# =============================================================================
# Mounts:
#   GET  /dashboard                       (static SPA — settings + preview)
#   POST /api/dashboard/login/request     (magic-link — optional if auth enabled)
#   GET  /api/dashboard/login/verify
#   GET  /api/dashboard/settings          (read — auth only if DASHBOARD_REQUIRE_AUTH=1)
#   PUT  /api/dashboard/settings          (save — auth only if DASHBOARD_REQUIRE_AUTH=1)
#   GET  /api/public/widget-settings      (called by chat-frame.html on load)
#   GET  /api/dashboard/health            (env sanity)
#
# Required when DASHBOARD_REQUIRE_AUTH=1:
#   DASHBOARD_ALLOWED_EMAILS=alice@yourcompany.com,bob@yourclient.com
#   DASHBOARD_SESSION_SECRET=               # 32+ random chars (HMAC key)
#   DASHBOARD_PUBLIC_BASE_URL=https://YOUR-API.up.railway.app
#
# Optional:
# DASHBOARD_REQUIRE_AUTH=                   # 1 / true = require cookie login for settings API (default: open / no login)
# DASHBOARD_SESSION_TTL_HOURS=168          # 1..720, default 168 (7 days)
# DASHBOARD_LINK_TTL_MINUTES=15            # 1..120, default 15
# DASHBOARD_FROM_EMAIL=                    # else same as transactional From: with Resend, RESEND_FROM before MAIL_FROM
# DASHBOARD_SETTINGS_COLLECTION=dashboard_settings
# DASHBOARD_TOKENS_COLLECTION=dashboard_login_tokens
# Published theme storage (dashboard "Make live" + GET /api/public/widget-settings):
# Default: file (JSON on disk; no Firestore required for theme). Optional override:
#   DASHBOARD_WIDGET_SETTINGS_BACKEND=firestore   # multi-replica hosts: shared Firestore
# Railway file mode: use 1 replica OR a volume mounted at DASHBOARD_WIDGET_SETTINGS_FILE, or saves differ per instance.
# DASHBOARD_WIDGET_SETTINGS_FILE=            # optional; default server/chatbot-api/data/widget-settings-store.json
#
# Optional: mirror "Make live" to a JSON file in a Git repo (GitHub Contents API).
# Matches what `published-widget-settings.json` in this repo expects. After push,
# serve that file next to chat-frame.html and set `<meta name="dfchat-published-widget-path" content="published-widget-settings.json" />`
# or `?publishedWidget=` on the iframe URL so visitors load theme from your CDN/site first.
# Requires a fine‑scoped token with repo Contents write (classic: `repo`; fine‑grained: Contents R/W).
# GITHUB_WIDGET_PUBLISH_TOKEN=
# GITHUB_WIDGET_PUBLISH_OWNER=your-org
# GITHUB_WIDGET_PUBLISH_REPO=your-repo
# GITHUB_WIDGET_PUBLISH_BRANCH=main
# GITHUB_WIDGET_PUBLISH_PATH=published-widget-settings.json
# DASHBOARD_PREVIEW_URL=https://qabot2026.github.io/testingone/chat-frame.html
#
# No mail provider / outbound blocked? When no provider is configured or the
# send fails, the magic sign-in link is printed to the server log instead.
# Open Railway → Logs, find the "==================== MAGIC LINK ============
# ========" block, and click the URL. Set DASHBOARD_PRINT_LOGIN_LINK=1 to
# print the link immediately (response returns fast; email send still runs in
# background when a provider is configured).
# DASHBOARD_PRINT_LOGIN_LINK=
#
# Wall-clock budget for one /login/request email send before falling back to
# logging the link. Default 25 seconds (5..120). Prevents a slow Gmail TLS or
# a hung HTTPS connection from blocking the dashboard sign-in spinner.
# DASHBOARD_LOGIN_EMAIL_TIMEOUT_MS=25000
# DASHBOARD_DEBUG=                         # 1 to log "magic-link sent via <provider>"
#

#   https://YOUR-API.up.railway.app/dashboard
